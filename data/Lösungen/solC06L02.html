public boolean zweiToken() {
while (!istAufToken()) {

zurückZumAnfang();
return false;
}
schritt();
}
//Einer wurde gefunden
if (!istVorWand()) {
schritt();

if (istVorWand()) {
zurückZumAnfang();
return false;
}
schritt();
}
} else {
zurückZumAnfang();
return false;
}
//Zwei wurden gefunden

if (!istVorWand()) {
schritt();

zurückZumAnfang();
return true;

}
} else {
zurückZumAnfang();
return true;
}
//Drei wurden gefunden. --> Falsch
zurückZumAnfang();
return false;
}
